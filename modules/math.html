🔵 核心定义

常用数学公式与算法：

import math

# ===== GCD / LCM =====
def gcd(a, b):
    return a if b == 0 else gcd(b, a % b)   # 欧几里得递推
    # Python: math.gcd(a, b)

def lcm(a, b):
    return a * b // gcd(a, b)               # lcm = a*b / gcd(a,b)

# ===== 快速幂（二分幂）=====
def power(base, exp, mod):
    result = 1
    base %= mod
    while exp > 0:
        if exp & 1:                          # 指数为奇数
            result = result * base % mod
        base = base * base % mod            # base平方
        exp >>= 1                           # 指数右移（÷2）
    return result
    # Python内置: pow(base, exp, mod)  # 同样O(log exp)，更快

# ===== 线性素数筛（欧拉筛）=====
def sieve(n):
    is_prime = [True] * (n + 1)
    primes = []
    is_prime[0] = is_prime[1] = False
    for i in range(2, n + 1):
        if is_prime[i]:
            primes.append(i)
        for p in primes:
            if i * p > n: break
            is_prime[i * p] = False         # 用i的最小质因子筛掉
            if i % p == 0: break            # 关键：保证每个合数只被筛一次
    return primes

# ===== 质因数分解（试除法）=====
def factorize(n):
    factors = []
    d = 2
    while d * d <= n:
        while n % d == 0:
            factors.append(d)
            n //= d
        d += 1
    if n > 1:
        factors.append(n)                   # 剩余的n是质数
    return factors

🟢 典型例题详解

例题1：计数质数（LeetCode 204）— 埃拉托色尼筛 
def countPrimes(n):
    if n < 2: return 0
    is_prime = [True] * n
    is_prime[0] = is_prime[1] = False
    # 只需筛到√n（大于√n的合数必有 ≤√n 的质因子）
    for i in range(2, int(n**0.5) + 1):
        if is_prime[i]:
            # 从 i² 开始筛（i×2,...,i×(i-1)已被更小的质数筛过）
            for j in range(i*i, n, i):
                is_prime[j] = False
    return sum(is_prime)

时间： O(n log log n)；空间： O(n)

例题2：数字 1 的个数（LeetCode 233）— 数位DP

题意： 给整数n，计算[1,n]所有数字中1出现的总次数。

def countDigitOne(n):
    count = 0
    factor = 1                   # 当前考虑的位（个位、十位、百位...）
    while factor <= n:
        # 将n分成三部分：高位 higher、当前位 current、低位 lower
        higher  = n // (factor * 10)
        current = (n // factor) % 10
        lower   = n % factor

        if current == 0:
            count += higher * factor            # 当前位为0：1出现次数=高位×factor
        elif current == 1:
            count += higher * factor + lower + 1  # 当前位为1：高位×factor+低位+1
        else:
            count += (higher + 1) * factor    # 当前位>1：(高位+1)×factor
        factor *= 10
    return count

时间： O(log n)；空间： O(1)

例题3：丑数 II（LeetCode 264）— 三指针

题意： 只含2、3、5质因子的数称为丑数，找第n个丑数。

def nthUglyNumber(n):
    dp = [1] * n                 # dp[i] = 第i+1个丑数
    p2 = p3 = p5 = 0            # 分别指向下一个乘以2/3/5的候选

    for i in range(1, n):
        next2 = dp[p2] * 2
        next3 = dp[p3] * 3
        next5 = dp[p5] * 5
        dp[i] = min(next2, next3, next5)   # 取最小的下一个丑数
        # 指针前进（可能有多个同时前进，避免重复）
        if dp[i] == next2: p2 += 1
        if dp[i] == next3: p3 += 1
        if dp[i] == next5: p5 += 1
    return dp[n-1]

时间： O(n)；空间： O(n)

🔴 易错点

① 快速幂取模时乘法溢出（Python不会溢出，但其他语言需注意）

❌ 错误（先乘后取模，中间可能溢出）

# C++/Java中：
result = result * base % mod  # base*result可能溢出64位

✅ 正确（Python自动大整数，但竞赛注意）

# Python pow(base, exp, mod) 内部处理好了
result = pow(base, exp, mod)
# 或：Python整数无溢出，直接 (a * b) % mod 安全

② 筛法下标范围错误

❌ 错误（从i×2开始筛）

for i in range(2, n):
    if is_prime[i]:
        for j in range(2*i, n, i):  # 从2i开始，效率低
            is_prime[j] = False     # 2i,...,i²-i都被更小的筛过了

✅ 正确（从i²开始筛）

for i in range(2, int(n**0.5)+1):
    if is_prime[i]:
        for j in range(i*i, n, i):  # 从i²开始，小的已被筛
            is_prime[j] = False

🔷 竞赛拓展

扩展欧几里得（exgcd）： 求 ax + by = gcd(a,b) 的整数解。用于求模逆元（当gcd=1时）。

def exgcd(a, b):
    if b == 0:
        return a, 1, 0                      # gcd, x, y
    g, x, y = exgcd(b, a % b)
    return g, y, x - (a // b) * y          # 辗转递推

def mod_inverse(a, mod):
    g, x, _ = exgcd(a, mod)
    if g != 1: return None                  # 无模逆元
    return x % mod                          # 确保正整数

中国剩余定理（CRT）： 求满足多个同余方程组的最小正整数解，要求模数两两互质。可合并多个分组的约束。

Lucas定理： 大模数下组合数取模 C(n,k) % p（p为质数）：分解n,k为p进制，逐位计算小组合数之积，O(p + log_p(n))。
