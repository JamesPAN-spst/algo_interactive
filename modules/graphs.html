BFS 模板（无权最短路）：

from collections import deque

def bfs(graph, start, end):
    queue = deque([(start, 0)])       # (当前节点, 距离)
    visited = {start}
    while queue:
        node, dist = queue.popleft()
        if node == end:
            return dist
        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor) # 入队时即标记，防重复
                queue.append((neighbor, dist + 1))
    return -1

Dijkstra 模板（非负权最短路）：

import heapq

def dijkstra(graph, start, n):
    dist = [float('inf')] * n
    dist[start] = 0
    heap = [(0, start)]              # (距离, 节点)

    while heap:
        d, u = heapq.heappop(heap)
        if d > dist[u]: continue     # 已找到更短路，跳过
        for v, w in graph[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                heapq.heappush(heap, (dist[v], v))
    return dist

例题1：岛屿数量（LeetCode 200）— DFS/BFS

题意： 给二维网格，'1'为陆地'0'为水，求岛屿数量（相连'1'的连通块数）。

思路： 遍历每个格子，遇到'1'进行DFS/BFS标记所有相连陆地，每次触发=一个岛屿。

def numIslands(grid):
    if not grid: return 0
    rows, cols = len(grid), len(grid[0])
    count = 0

    def dfs(r, c):
        # 越界或是水，停止
        if r < 0 or r >= rows or c < 0 or c >= cols or grid[r][c] != '1':
            return
        grid[r][c] = '0'        # 标记已访问（原地修改）
        dfs(r+1,c); dfs(r-1,c); dfs(r,c+1); dfs(r,c-1)

    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == '1':
                count += 1       # 发现新岛屿
                dfs(r, c)        # 淹没整个岛屿
    return count

时间： O(m×n)；空间： O(m×n)（递归栈）

例题2：课程表（LeetCode 207）— 拓扑排序（Kahn算法）

题意： n门课程，有前置要求，判断能否完成所有课程。

思路： 建有向图，BFS拓扑排序，若存在环则无法完成。

from collections import deque

def canFinish(numCourses, prerequisites):
    graph = [[] for _ in range(numCourses)]
    in_degree = [0] * numCourses

    for a, b in prerequisites:         # b → a（先修b才能学a）
        graph[b].append(a)
        in_degree[a] += 1

    queue = deque([i for i in range(numCourses) if in_degree[i] == 0])
    completed = 0

    while queue:
        node = queue.popleft()
        completed += 1                 # 完成该课程
        for neighbor in graph[node]:
            in_degree[neighbor] -= 1
            if in_degree[neighbor] == 0:
                queue.append(neighbor) # 前置都完成了，可学

    return completed == numCourses     # 全部完成则无环

时间： O(V+E)；空间： O(V+E)

例题3：网络延迟时间（LeetCode 743）— Dijkstra

题意： n个节点有向加权图，从k出发，求所有节点收到信号的最短时间。

import heapq
from collections import defaultdict

def networkDelayTime(times, n, k):
    graph = defaultdict(list)
    for u, v, w in times:
        graph[u].append((v, w))        # 有向带权图

    dist = {i: float('inf') for i in range(1, n+1)}
    dist[k] = 0
    heap = [(0, k)]                    # (距离, 节点)

    while heap:
        d, u = heapq.heappop(heap)
        if d > dist[u]: continue       # 旧路径，跳过
        for v, w in graph[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                heapq.heappush(heap, (dist[v], v))

    max_dist = max(dist.values())
    return max_dist if max_dist < float('inf') else -1

时间： O(E log V)；空间： O(V+E)

🔴 易错点

① visited 标记时机错误导致重复入队

❌ 错误（出队时才标记）

while queue:
    node = queue.popleft()
    visited.add(node)          # 出队才标记，太晚了！
    for nb in graph[node]:
        if nb not in visited:
            queue.append(nb)   # 同一节点可能多次入队

✅ 正确（入队时即标记）

visited = {start}
while queue:
    node = queue.popleft()
    for nb in graph[node]:
        if nb not in visited:
            visited.add(nb)    # 入队时立即标记
            queue.append(nb)

② 无向图只添加单向边

❌ 错误（单向）

for u, v in edges:
    graph[u].append(v)  # 只有u→v
    # 忘记 graph[v].append(u)

✅ 正确（双向）

for u, v in edges:
    graph[u].append(v)  # u→v
    graph[v].append(u)  # v→u（无向图必须）

③ Dijkstra用于含负权边

❌ 错误（负权用Dijkstra）

# 存在边权 -5
# Dijkstra贪心选最小，但负权边
# 可能让后来的路径更短，得到错误答案

✅ 正确（负权用Bellman-Ford）

# Bellman-Ford: 松弛V-1次
for _ in range(n - 1):
    for u, v, w in edges:
        if dist[u] + w < dist[v]:
            dist[v] = dist[u] + w

🔷 竞赛拓展

A* 搜索： Dijkstra + 启发式函数 h(n)（如曼哈顿距离），优先扩展 f(n)=g(n)+h(n) 最小的节点，比Dijkstra快但需要可接受启发式。

强连通分量（SCC）： Kosaraju/Tarjan 算法，O(V+E)。用于缩点、DAG转化。

欧拉路径： 一笔画问题。无向图：所有节点度数为偶数（欧拉回路）或恰好2个奇数度节点（欧拉路径）。

# Hierholzer算法求欧拉路径（O(E)）
from collections import defaultdict

def findItinerary(tickets):
    graph = defaultdict(list)
    for src, dst in sorted(tickets, reverse=True):
        graph[src].append(dst)   # 逆序排序，pop()取字典序最小
    result = []
    def dfs(node):
        while graph[node]:
            dfs(graph[node].pop())   # 深度优先
        result.append(node)          # 后序加入即为逆序路径
    dfs("JFK")
    return result[::-1]
