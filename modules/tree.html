🔵 核心定义

树节点定义与三种DFS遍历：

class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

def preorder(root, result=[]):    # 前序：根-左-右
    if not root: return
    result.append(root.val)       # 先处理根
    preorder(root.left, result)
    preorder(root.right, result)

def inorder(root, result=[]):     # 中序：左-根-右
    if not root: return
    inorder(root.left, result)
    result.append(root.val)       # 中间处理根（BST中序=升序）
    inorder(root.right, result)

def postorder(root, result=[]):   # 后序：左-右-根
    if not root: return
    postorder(root.left, result)
    postorder(root.right, result)
    result.append(root.val)       # 最后处理根（适合删除节点/计算高度）

BST核心性质： 对于任意节点，左子树所有节点 < 当前节点 < 右子树所有节点。中序遍历BST得到升序序列。

🟢 典型例题详解

例题1：二叉树的最大深度（LeetCode 104）

题意： 求二叉树的最大深度。

def maxDepth(root):
    if not root: return 0              # 空节点深度为0
    left_depth  = maxDepth(root.left)  # 左子树深度
    right_depth = maxDepth(root.right) # 右子树深度
    return 1 + max(left_depth, right_depth)  # 加上当前层

# 迭代BFS版本
from collections import deque
def maxDepth_bfs(root):
    if not root: return 0
    depth = 0
    queue = deque([root])
    while queue:
        depth += 1                     # 每处理完一层，深度+1
        for _ in range(len(queue)):    # 处理当前层所有节点
            node = queue.popleft()
            if node.left:  queue.append(node.left)
            if node.right: queue.append(node.right)
    return depth

时间： O(n)；空间： O(h)，h为树高

例题2：验证二叉搜索树（LeetCode 98）

题意： 验证给定二叉树是否为有效的BST。

def isValidBST(root):
    def validate(node, min_val, max_val):
        if not node: return True
        # 当前节点必须在(min_val, max_val)范围内
        if node.val <= min_val or node.val >= max_val:
            return False
        # 左子树上界更新为当前值，右子树下界更新为当前值
        return (validate(node.left,  min_val, node.val) and
                validate(node.right, node.val, max_val))

    return validate(root, float('-inf'), float('inf'))

# 或：利用中序遍历严格递增
def isValidBST_inorder(root):
    prev = float('-inf')
    def inorder(node):
        nonlocal prev
        if not node: return True
        if not inorder(node.left): return False
        if node.val <= prev: return False   # 中序不严格递增→非BST
        prev = node.val
        return inorder(node.right)
    return inorder(root)

时间： O(n)；空间： O(h)

例题3：二叉树的层序遍历（LeetCode 102） 
from collections import deque

def levelOrder(root):
    if not root: return []
    result = []
    queue = deque([root])
    while queue:
        level = []
        for _ in range(len(queue)):        # 逐层处理
            node = queue.popleft()
            level.append(node.val)
            if node.left:  queue.append(node.left)
            if node.right: queue.append(node.right)
        result.append(level)               # 每层结果单独收集
    return result

时间： O(n)；空间： O(n)（最宽层的节点数）

🔴 易错点

① 递归终止条件遗漏

❌ 错误（缺少空节点判断）

def maxDepth(root):
    # 忘记 if not root: return 0
    left = maxDepth(root.left)   # root可能为None，报错
    right = maxDepth(root.right)
    return 1 + max(left, right)

✅ 正确（先判空）

def maxDepth(root):
    if not root: return 0        # 空节点返回0
    left = maxDepth(root.left)
    right = maxDepth(root.right)
    return 1 + max(left, right)

② BST验证时只检查直接父子关系

❌ 错误（只比较父子）

def isValid(root):
    if not root: return True
    if root.left and root.left.val >= root.val: return False
    if root.right and root.right.val <= root.val: return False
    return isValid(root.left) and isValid(root.right)
    # 反例: [5,4,6,null,null,3,7] 这个会通过但不是BST
    # 因为3在右子树6的左边，3 < 5（根）违法

✅ 正确（传递上下界）

def isValid(node, lo, hi):
    if not node: return True
    if not (lo < node.val < hi): return False
    return (isValid(node.left, lo, node.val) and
            isValid(node.right, node.val, hi))

🔷 竞赛拓展

AVL树： 自平衡BST，每个节点左右子树高度差 ≤ 1。插入/删除后通过旋转保持平衡。4种旋转：LL/RR（单旋）、LR/RL（双旋）。

红黑树： Python内置 SortedList 底层实现。插入/删除/查找均 O(log n)，常数较小。5条性质保证高度 ≤ 2log(n+1)。

Morris遍历： O(1)空间的中序遍历，利用叶节点空指针作线索。

def morrisInorder(root):
    result = []
    cur = root
    while cur:
        if not cur.left:
            result.append(cur.val)    # 左子树为空，直接访问
            cur = cur.right
        else:
            # 找中序前驱（左子树最右节点）
            prev = cur.left
            while prev.right and prev.right != cur:
                prev = prev.right
            if not prev.right:
                prev.right = cur      # 建立线索
                cur = cur.left
            else:
                prev.right = None     # 删除线索
                result.append(cur.val)
                cur = cur.right
    return result
