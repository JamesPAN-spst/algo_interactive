🔵 核心定义

Trie完整实现（字典节点）：

class TrieNode:
    def __init__(self):
        self.children = {}             # 字符 → TrieNode（或用数组[None]*26）
        self.is_end = False            # 是否是某个单词的结尾

class Trie:
    def __init__(self):
        self.root = TrieNode()

    def insert(self, word):
        node = self.root
        for ch in word:
            if ch not in node.children:
                node.children[ch] = TrieNode()   # 节点不存在则创建
            node = node.children[ch]
        node.is_end = True             # 标记单词结束

    def search(self, word):
        node = self.root
        for ch in word:
            if ch not in node.children:
                return False           # 字符不存在
            node = node.children[ch]
        return node.is_end             # 必须是完整单词（not just前缀）

    def startsWith(self, prefix):
        node = self.root
        for ch in prefix:
            if ch not in node.children:
                return False
            node = node.children[ch]
        return True                    # 只要前缀存在即可

数组优化（固定字符集时更快）：

class TrieNode:
    def __init__(self):
        self.children = [None] * 26    # 只含小写字母时
        self.is_end = False

def insert(self, word):
    node = self.root
    for ch in word:
        idx = ord(ch) - ord('a')       # 字符→下标 O(1)
        if not node.children[idx]:
            node.children[idx] = TrieNode()
        node = node.children[idx]
    node.is_end = True

例题1：实现Trie前缀树（LeetCode 208）

题意： 实现Trie，支持insert、search、startsWith三种操作。

思路： 如上面核心定义，用字典存children，is_end标记单词结尾。

# 完整可运行版本（与核心定义相同）
class Trie:
    def __init__(self):
        self.root = {}              # 用嵌套字典简化

    def insert(self, word):
        node = self.root
        for ch in word:
            node = node.setdefault(ch, {})   # 不存在则创建
        node['#'] = True            # '#'作为结束标记

    def search(self, word):
        node = self.root
        for ch in word:
            if ch not in node: return False
            node = node[ch]
        return '#' in node          # 必须有结束标记

    def startsWith(self, prefix):
        node = self.root
        for ch in prefix:
            if ch not in node: return False
            node = node[ch]
        return True
例题2：单词搜索 II（LeetCode 212）— Trie + DFS回溯

题意： 在二维字母网格中搜索给定单词列表中所有能找到的单词。

def findWords(board, words):
    # 构建Trie
    trie = {}
    for word in words:
        node = trie
        for ch in word:
            node = node.setdefault(ch, {})
        node['$'] = word            # '$'存原单词

    rows, cols = len(board), len(board[0])
    result = []

    def dfs(r, c, node):
        ch = board[r][c]
        if ch not in node: return
        next_node = node[ch]
        if '$' in next_node:
            result.append(next_node['$'])   # 找到单词
            del next_node['$']              # 防重复收集

        board[r][c] = '#'                   # 标记已访问
        for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < rows and 0 <= nc < cols and board[nr][nc] != '#':
                dfs(nr, nc, next_node)
        board[r][c] = ch                    # 回溯

        # 剪枝：若节点为空则删除（减少无效遍历）
        if not next_node:
            del node[ch]

    for r in range(rows):
        for c in range(cols):
            dfs(r, c, trie)
    return result

时间： O(M × 4 × 3^(L-1))，M为格子数，L为最长单词长度；空间： O(∑|word|)

例题3：键值映射（LeetCode 677）— 前缀和Trie

题意： 实现MapSum，支持键值插入和前缀求和。

class MapSum:
    def __init__(self):
        self.trie = {}
        self.vals = {}              # 存每个key的值

    def insert(self, key, val):
        delta = val - self.vals.get(key, 0)  # 增量（覆盖时用差值）
        self.vals[key] = val
        node = self.trie
        for ch in key:
            node = node.setdefault(ch, {'_sum': 0})
            node['_sum'] += delta   # 每个节点累加经过该前缀的总和

    def sum(self, prefix):
        node = self.trie
        for ch in prefix:
            if ch not in node: return 0
            node = node[ch]
        return node.get('_sum', 0)  # 直接读取前缀和

🔴 易错点

① 搜索时混淆search和startsWith

❌ 错误（search只检查路径存在）

def search(self, word):
    node = self.root
    for ch in word:
        if ch not in node.children: return False
        node = node.children[ch]
    return True    # 路径存在就返回True
    # 错！"app"路径存在但如果只插入了"apple"，search("app")应返回False

✅ 正确（必须检查is_end）

def search(self, word):
    node = self.root
    for ch in word:
        if ch not in node.children: return False
        node = node.children[ch]
    return node.is_end  # 必须是完整单词！

② 字符映射越界（字符集未统一）

❌ 错误（大小写混用）

idx = ord(ch) - ord('a')    # 转换大写字母会得到负数！
# 'A' - 'a' = -32  → 数组越界

✅ 正确（统一处理）

ch = ch.lower()             # 先统一小写
idx = ord(ch) - ord('a')    # 再转换，范围[0,25]
# 或用字典代替数组，支持所有字符

🔷 竞赛拓展

压缩Trie（Radix Tree）： 合并只有一个子节点的链式路径为单条边，大幅节省节点数量。Go的路由库广泛使用此结构。

AC自动机： 在Trie基础上添加失配指针（类似KMP的fail数组），实现多模式字符串同时匹配，O(n + ∑pattern + 匹配数)。

01-Trie（异或Trie）： 将整数按二进制位建Trie（从高位到低位），O(log V)查找与某数异或最大/最小的数。

class XorTrie:
    def __init__(self):
        self.root = [None, None]   # 0或1

    def insert(self, num):
        node = self.root
        for i in range(31, -1, -1):   # 从高位到低位
            bit = (num >> i) & 1
            if not node[bit]:
                node[bit] = [None, None]
            node = node[bit]

    def maxXor(self, num):
        node = self.root
        xor = 0
        for i in range(31, -1, -1):
            bit = (num >> i) & 1
            want = 1 - bit             # 希望取相反位（最大化XOR）
            if node[want]:
                xor |= (1 << i)
                node = node[want]
            else:
                node = node[bit]
        return xor
