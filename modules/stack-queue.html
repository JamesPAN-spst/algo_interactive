🔵 核心定义

单调栈模板（求下一个更大元素）：

def nextGreaterElement(nums):
    n = len(nums)
    result = [-1] * n
    stack = []                         # 单调递减栈（存下标）

    for i in range(n):
        # 当前元素比栈顶大 → 栈顶元素的"下一个更大"就是当前元素
        while stack and nums[i] > nums[stack[-1]]:
            idx = stack.pop()
            result[idx] = nums[i]      # 找到了idx的下一个更大元素
        stack.append(i)
    return result

最小栈（O(1)获取最小值）：

class MinStack:
    def __init__(self):
        self.stack = []
        self.min_stack = []            # 辅助栈：存当前最小值

    def push(self, val):
        self.stack.append(val)
        # 辅助栈记录当前最小值（只有更小时才更新）
        min_val = min(val, self.min_stack[-1] if self.min_stack else val)
        self.min_stack.append(min_val)

    def pop(self):
        self.stack.pop()
        self.min_stack.pop()           # 同步弹出

    def getMin(self):
        return self.min_stack[-1]      # O(1) 取最小

🟢 典型例题详解

例题1：有效的括号（LeetCode 20）

题意： 判断括号字符串是否有效（括号匹配且顺序正确）。

def isValid(s):
    stack = []
    mapping = {')': '(', '}': '{', ']': '['}  # 右括号→对应左括号

    for char in s:
        if char in mapping:           # 遇到右括号
            top = stack.pop() if stack else '#'  # 栈空则用'#'占位
            if mapping[char] != top:
                return False          # 不匹配
        else:
            stack.append(char)        # 左括号入栈
    return not stack                  # 栈空说明全部匹配

时间： O(n)；空间： O(n)

例题2：柱状图中最大矩形（LeetCode 84）— 单调栈

题意： 给柱状图高度数组，找能构成的最大矩形面积。

def largestRectangleArea(heights):
    heights = heights + [0]            # 末尾加0，强制清空栈
    stack = [-1]                       # 初始放哨兵
    result = 0

    for i, h in enumerate(heights):
        while stack[-1] != -1 and heights[stack[-1]] >= h:
            height = heights[stack.pop()]      # 弹出作为矩形高度
            width = i - stack[-1] - 1          # 宽度 = 当前位置 - 新栈顶 - 1
            result = max(result, height * width)
        stack.append(i)
    return result

时间： O(n)；空间： O(n)

例题3：用两个栈实现队列（LeetCode 232） 
class MyQueue:
    def __init__(self):
        self.stack_in  = []   # 输入栈：接收push
        self.stack_out = []   # 输出栈：提供pop/peek

    def push(self, x):
        self.stack_in.append(x)

    def _transfer(self):
        if not self.stack_out:                 # 输出栈空才转移
            while self.stack_in:
                self.stack_out.append(self.stack_in.pop())

    def pop(self):
        self._transfer()
        return self.stack_out.pop()

    def peek(self):
        self._transfer()
        return self.stack_out[-1]

    def empty(self):
        return not self.stack_in and not self.stack_out

摊销： push O(1)，pop O(1) 摊销（每个元素最多转一次）；空间： O(n)

🔴 易错点

① 用list的pop(0)而不是deque实现队列

❌ 错误（O(n)操作）

queue = []
queue.append(x)      # 入队 O(1)
queue.pop(0)         # 出队 O(n)！列表头部删除需移动所有元素

✅ 正确（deque，O(1)）

from collections import deque
queue = deque()
queue.append(x)      # 入队 O(1)
queue.popleft()      # 出队 O(1)

② 单调栈比较条件写错（>= vs >）

❌ 错误（求下一个更大用>=）

while stack and nums[i] >= nums[stack[-1]]:
    # >= 会把相等元素也弹出
    # 对于重复元素会产生错误答案
    result[stack.pop()] = nums[i]

✅ 正确（严格大于用>）

while stack and nums[i] > nums[stack[-1]]:
    # 严格大于，相等的留在栈里
    result[stack.pop()] = nums[i]

🔷 竞赛拓展

单调队列优化DP： 当状态转移 dp[i] = min/max(dp[j]) for j in [i-k, i-1] 时，用单调队列代替线性查找，将O(nk)降至O(n)。

from collections import deque

def maxSlidingWindow(nums, k):
    dq = deque()           # 存下标，维护单调递减
    result = []
    for i, num in enumerate(nums):
        if dq and dq[0] < i - k + 1:  # 移除超出窗口的旧下标
            dq.popleft()
        while dq and nums[dq[-1]] < num:  # 保持队列单调递减
            dq.pop()
        dq.append(i)
        if i >= k - 1:
            result.append(nums[dq[0]])    # 队首是窗口最大值下标
    return result

表达式求值（中缀→后缀）： 双栈法（操作符栈+操作数栈）或逆波兰表达式法，支持优先级和括号处理。
