🔵 核心定义

Dinic算法（最高效的实用最大流算法）：

from collections import deque

class MaxFlow:
    def __init__(self, n):
        self.n = n
        self.graph = [[] for _ in range(n)]

    def add_edge(self, u, v, cap):
        # 关键：用成对边（正向+反向）存储，反向边初始容量为0
        self.graph[u].append([v, cap, len(self.graph[v])])
        self.graph[v].append([u, 0,   len(self.graph[u])-1])

    def bfs(self, s, t):
        """BFS构建层次图，返回是否存在s到t的路径"""
        self.level = [-1] * self.n
        self.level[s] = 0
        queue = deque([s])
        while queue:
            u = queue.popleft()
            for v, cap, _ in self.graph[u]:
                if cap > 0 and self.level[v] < 0:
                    self.level[v] = self.level[u] + 1
                    queue.append(v)
        return self.level[t] >= 0

    def dfs(self, u, t, f, iter_):
        """DFS在层次图上找增广路，iter_是当前弧优化指针"""
        if u == t: return f
        while iter_[u] < len(self.graph[u]):
            v, cap, rev = self.graph[u][iter_[u]]
            if cap > 0 and self.level[v] == self.level[u] + 1:
                d = self.dfs(v, t, min(f, cap), iter_)
                if d > 0:
                    self.graph[u][iter_[u]][1] -= d   # 减少正向容量
                    self.graph[v][rev][1] += d         # 增加反向容量
                    return d
            iter_[u] += 1                             # 当前弧无效，跳到下一条
        return 0

    def max_flow(self, s, t):
        flow = 0
        while self.bfs(s, t):                        # 反复构建层次图
            iter_ = [0] * self.n                     # 每轮重置当前弧指针
            while True:
                f = self.dfs(s, t, float('inf'), iter_)
                if f == 0: break
                flow += f
        return flow

最大流最小割定理： 从s到t的最大流 = 分离s和t的最小割容量。

📊 复杂度速查

算法时间复杂度适用场景
Ford-Fulkerson（DFS增广）O(E × max_flow)容量小的图
Edmonds-Karp（BFS增广）O(V × E²)通用最大流
Dinic（层次图+当前弧）O(V² × E)实用中最快
Dinic（单位容量图）O(E × √V)二分图最大匹配
最小费用最大流（SPFA+增广）O(V × E × flow)有代价的流问题

🖼 结构图示：残量网络与增广路

原始网络（容量）:
    s --10--> A --9--> t
    s --8---> B --7--> t
         A --6--> B

找增广路：s→A→t (flow=9)
残量网络更新:
    s --1--> A    (正向剩余10-9=1)
    A --9--> s    (反向增加9，允许"撤销")
    A --0--> t    (正向剩余9-9=0)
    t --9--> A    (反向增加9)

再找：s→B→t (flow=7)
再找：s→A→B→t (flow=1，利用A→B=6)

总最大流 = 9+7+1 = 17

最小割 = 去掉 (A→t容量0) + (B→t容量7): 容量 = 9+7=16
实际最大流 = 最小割容量 (最大流最小割定理!)

例题1：二分图最大匹配（匈牙利算法）

题意： 左右两侧节点，给定可匹配边集，求最大匹配数。

def maxBipartiteMatching(graph, left_n, right_n):
    """graph[u] = 左侧u节点可以匹配的右侧节点列表"""
    match = [-1] * right_n     # match[v] = 右侧v当前匹配的左侧节点

    def dfs(u, visited):
        for v in graph[u]:
            if not visited[v]:
                visited[v] = True
                # v未被匹配，或v当前匹配的节点可以改配
                if match[v] == -1 or dfs(match[v], visited):
                    match[v] = u
                    return True
        return False

    result = 0
    for u in range(left_n):
        visited = [False] * right_n
        if dfs(u, visited):
            result += 1
    return result

# 等价地，可用Dinic最大流建模（更快）
# 建图：超源s→左节点(容量1)，左→右(容量1)，右→超汇t(容量1)

时间： O(V × E)；Dinic建模后 O(E × √V)

例题2：项目选择 / 最小割建模

题意： 选某些项目获利，但选项目需选其依赖，问最大净获利。

"""
建模思路（最大权闭合子图 → 最大流）:
- 超源s连向所有正收益项目，边容量=收益
- 所有负收益项目连向超汇t，边容量=|收益|
- 项目依赖关系建无限容量边（a依赖b则a→b，容量∞）
- 最大净收益 = 正收益之和 - 最小割

最大净收益 = 正收益总和 - min_cut(s, t)
"""

def maxProfit(projects, dependencies):
    INF = float('inf')
    n = len(projects)
    mf = MaxFlow(n + 2)        # +2 for s=n and t=n+1
    s, t = n, n + 1
    pos_sum = 0

    for i, profit in enumerate(projects):
        if profit > 0:
            mf.add_edge(s, i, profit)    # s连正收益项目
            pos_sum += profit
        else:
            mf.add_edge(i, t, -profit)  # 负收益项目连t

    for u, v in dependencies:
        mf.add_edge(u, v, INF)          # 依赖关系（无限容量）

    return pos_sum - mf.max_flow(s, t)
例题3：最小费用最大流（MCMF）框架 
from collections import defaultdict
import heapq

def mcmf(graph, s, t, n):
    """最小费用最大流：Bellman-Ford/SPFA找最短增广路"""
    total_flow = 0
    total_cost = 0

    while True:
        # SPFA找最短（费用最小）增广路
        dist = [float('inf')] * n
        in_queue = [False] * n
        prev_node = [-1] * n
        prev_edge = [-1] * n
        dist[s] = 0
        queue = deque([s])
        in_queue[s] = True

        while queue:
            u = queue.popleft()
            in_queue[u] = False
            for i, (v, cap, cost) in enumerate(graph[u]):
                if cap > 0 and dist[u] + cost < dist[v]:
                    dist[v] = dist[u] + cost
                    prev_node[v] = u
                    prev_edge[v] = i
                    if not in_queue[v]:
                        queue.append(v)
                        in_queue[v] = True

        if dist[t] == float('inf'): break   # 无增广路

        # 找路径最小容量
        flow = float('inf')
        v = t
        while v != s:
            u = prev_node[v]
            flow = min(flow, graph[u][prev_edge[v]][1])
            v = u

        # 更新残量
        v = t
        while v != s:
            u = prev_node[v]
            graph[u][prev_edge[v]][1] -= flow
            # 更新反向边...
            v = u

        total_flow += flow
        total_cost += flow * dist[t]

    return total_flow, total_cost

🔴 易错点

① 反向边初始容量漏加

❌ 错误（忘记反向边）

graph[u].append([v, cap])
# 没有加反向边
# 算法无法通过"撤销"找到更优路径

✅ 正确（正向+反向成对）

graph[u].append([v, cap, len(graph[v])])    # 正向
graph[v].append([u, 0, len(graph[u])-1])   # 反向（容量0）
# 两边通过下标互相引用

② Dinic的层次图重复BFS没有重置level

❌ 错误（沿用旧level）

def max_flow(self, s, t):
    while True:
        # 忘记重建层次图（BFS）
        f = self.dfs(s, t, INF, iter_)
        if f == 0: break
        flow += f

✅ 正确（每轮重新BFS）

def max_flow(self, s, t):
    flow = 0
    while self.bfs(s, t):      # 重建层次图
        iter_ = [0] * self.n   # 重置当前弧
        while f := self.dfs(s, t, INF, iter_):
            flow += f
    return flow

🔷 竞赛拓展

最小费用最大流（MCMF）： 在满足最大流的条件下，使总费用最小。SPFA找最短费用路径，结合Ford-Fulkerson增广，时间 O(V × E × flow)。

上下界可行流： 每条边有容量上界和下界，求满足约束的可行流。转化为无下界网络的最大流问题。

全局最小割（Stoer-Wagner）： 不指定源汇的最小割，O(V³)或O(VE + V²log V)。
