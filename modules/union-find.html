🔵 核心定义

并查集完整实现（路径压缩 + 按秩合并）：

class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))   # 初始每个节点是自己的根
        self.rank   = [0] * n          # 树的高度估计（按秩合并用）
        self.count  = n                # 连通分量数量

    def find(self, x):
        # 路径压缩：递归找根，并将沿途所有节点直接连到根
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])  # 关键：压缩路径
        return self.parent[x]

    def union(self, x, y):
        rx, ry = self.find(x), self.find(y)
        if rx == ry: return False       # 已在同一集合
        # 按秩合并：把秩小的树接到秩大的树下面
        if self.rank[rx] < self.rank[ry]:
            rx, ry = ry, rx
        self.parent[ry] = rx            # ry的根挂到rx下
        if self.rank[rx] == self.rank[ry]:
            self.rank[rx] += 1          # 只有两树等高时，新树高度+1
        self.count -= 1
        return True

    def connected(self, x, y):
        return self.find(x) == self.find(y)

摊销时间复杂度： 路径压缩 + 按秩合并后，每次 find/union 操作的摊销时间为 O(α(n))（Ackermann逆函数），实际上极接近 O(1)。

🖼 结构图示：路径压缩过程

【未压缩的深层树】
  0←1←2←3←4    find(4): 0←1←2←3←4 需要遍历5步

【路径压缩后】
  parent[4]=find(3)=find(2)=find(1)=find(0)=0
  所有节点直接指向根：
       0
     / | \ \
    1  2  3  4   下次find任意节点只需1步！

【按秩合并示意】
  集合A: 0←1←2 (rank=2)
  集合B:    3   (rank=0)
  union(2,3): rank[A]>rank[B], 将B接到A:
       0←1←2←3   rank不变
  
  如果rank相等:
  集合A: 0←1 (rank=1)
  集合B: 2←3 (rank=1)
  union: 3接到1下，新rank=2
       0←1←3←2

🟢 典型例题详解

例题1：朋友圈数量（LeetCode 547）

题意： n个学生，isConnected[i][j]=1表示i和j是朋友（传递），求朋友圈总数。

def findCircleNum(isConnected):
    n = len(isConnected)
    uf = UnionFind(n)
    for i in range(n):
        for j in range(i+1, n):
            if isConnected[i][j]:
                uf.union(i, j)     # 合并为同一朋友圈
    return uf.count                # 剩余连通分量数即为朋友圈数

时间： O(n² α(n))；空间： O(n)

例题2：冗余连接（LeetCode 684）— 找成环的边

题意： 无向图中有n个节点和n条边（树+1条多余边），找出多余的边。

def findRedundantConnection(edges):
    n = len(edges)
    uf = UnionFind(n + 1)          # 节点从1开始
    for u, v in edges:
        if not uf.union(u, v):     # union返回False = 已连通 = 成环
            return [u, v]          # 这条边就是冗余边
    return []

时间： O(n α(n))；空间： O(n)

例题3：账户合并（LeetCode 721）— 多键并查集

题意： 多个账户，每个有邮箱列表，合并相同邮箱账户。

from collections import defaultdict

def accountsMerge(accounts):
    email_to_id = {}        # 邮箱→代表账户ID
    uf_parent = {}

    def find(x):
        if uf_parent.setdefault(x, x) != x:
            uf_parent[x] = find(uf_parent[x])
        return uf_parent[x]

    def union(x, y):
        uf_parent[find(x)] = find(y)

    # 将同账户下所有邮箱union在一起
    for account in accounts:
        name = account[0]
        first_email = account[1]
        for email in account[1:]:
            if email not in email_to_id:
                email_to_id[email] = first_email    # 用第一个邮箱代表
            union(email, first_email)               # 合并到第一个邮箱

    # 收集每个根代表的邮箱集合
    root_to_emails = defaultdict(list)
    for email in email_to_id:
        root_to_emails[find(email)].append(email)

    # 组合结果（邮箱对应账户名）
    result = []
    email_to_name = {e: accounts[i][0] for i, acc in enumerate(accounts) for e in acc[1:]}
    for root, emails in root_to_emails.items():
        result.append([email_to_name[root]] + sorted(emails))
    return result

🔴 易错点

① find未路径压缩导致树退化

❌ 错误（无压缩）

def find(x):
    while parent[x] != x:
        x = parent[x]    # 每次都遍历整条链，O(n)
    return x

✅ 正确（路径压缩）

def find(x):
    if parent[x] != x:
        parent[x] = find(parent[x])  # 递归压缩
    return parent[x]

② union方向混乱（不按秩）

❌ 错误（总是ry→rx，不管高度）

def union(x, y):
    rx, ry = find(x), find(y)
    parent[ry] = rx    # 可能把高树接到矮树下
    # 导致树退化，压缩效果差

✅ 正确（矮挂高）

def union(x, y):
    rx, ry = find(x), find(y)
    if rank[rx] < rank[ry]: rx, ry = ry, rx
    parent[ry] = rx    # 秩小的挂到秩大的下面
    if rank[rx] == rank[ry]: rank[rx] += 1

🔷 竞赛拓展

带权并查集： 在 parent 数组基础上维护 weight 数组，记录节点到根的权值关系。用于区间相等/比例关系维护。

class WeightedUF:
    def __init__(self, n):
        self.parent = list(range(n))
        self.weight = [1.0] * n    # weight[i] = i到parent[i]的比值

    def find(self, x):
        if self.parent[x] != x:
            root = self.find(self.parent[x])
            self.weight[x] *= self.weight[self.parent[x]]  # 更新权值
            self.parent[x] = root
        return self.parent[x]

    def union(self, x, y, val):    # val = x/y
        rx, ry = self.find(x), self.find(y)
        if rx == ry: return
        self.parent[rx] = ry
        self.weight[rx] = val * self.weight[y] / self.weight[x]

离线查询（Kruskal重构树）： 处理"最小瓶颈路径"问题，将边按权值排序后建树，点对的LCA即为最小瓶颈。

并查集 + 时间回退（可撤销并查集）： 按秩合并（不路径压缩），用栈记录每次操作前的状态，支持O(log n)回滚。
