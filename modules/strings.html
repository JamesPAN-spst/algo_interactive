🔵 核心定义

KMP算法 — 前缀函数（failure function）构建：

def build_kmp(pattern):
    n = len(pattern)
    fail = [0] * n          # fail[i] = pattern[:i+1] 的最长真前缀=后缀的长度
    j = 0
    for i in range(1, n):
        while j > 0 and pattern[i] != pattern[j]:
            j = fail[j-1]   # 回退：利用已匹配信息
        if pattern[i] == pattern[j]:
            j += 1
        fail[i] = j
    return fail

def kmp_search(text, pattern):
    fail = build_kmp(pattern)
    j = 0
    results = []
    for i in range(len(text)):
        while j > 0 and text[i] != pattern[j]:
            j = fail[j-1]   # 失配时回退到最长匹配前缀
        if text[i] == pattern[j]:
            j += 1
        if j == len(pattern):
            results.append(i - j + 1)   # 匹配成功，记录起始位置
            j = fail[j-1]               # 继续搜索
    return results

Rabin-Karp 滚动哈希：

def rabin_karp(text, pattern):
    BASE, MOD = 31, 10**9 + 7
    n, m = len(text), len(pattern)
    if m > n: return []

    # 计算pattern的哈希值
    p_hash = 0
    t_hash = 0
    power = 1
    for i in range(m):
        p_hash = (p_hash * BASE + ord(pattern[i])) % MOD
        t_hash = (t_hash * BASE + ord(text[i])) % MOD
        if i > 0: power = (power * BASE) % MOD

    results = []
    if p_hash == t_hash and text[:m] == pattern:
        results.append(0)
    for i in range(1, n - m + 1):
        # 滚动更新哈希：去掉旧字符，加入新字符
        t_hash = (t_hash - ord(text[i-1]) * power % MOD + MOD) % MOD
        t_hash = (t_hash * BASE + ord(text[i+m-1])) % MOD
        if t_hash == p_hash and text[i:i+m] == pattern:
            results.append(i)
    return results

fail数组构建:
  i=0: fail[0]=0  (初始)
  i=1: b≠a, fail[1]=0
  i=2: c≠a, fail[2]=0
  i=3: a=a, fail[3]=1
  i=4: b=b, fail[4]=2
  i=5: d≠c, j回退=fail[1]=0, d≠a, fail[5]=0

fail = [0,0,0,1,2,0]

匹配过程（文本位置5，c≠d失配时）:
  已匹配 "abcab"，最长公共前后缀长度=fail[4]=2
  → j跳回2（已匹配"ab"），继续从text[5]='c'与pattern[2]='c'比较
  → 匹配继续，不需要回退文本指针！节省了O(n)时间

例题1：字母异位词分组（LeetCode 49）— 哈希

题意： 将字符串列表按字母异位词分组。

from collections import defaultdict

def groupAnagrams(strs):
    groups = defaultdict(list)
    for s in strs:
        key = tuple(sorted(s))         # 排序后相同→同一组
        groups[key].append(s)
    return list(groups.values())

# 优化版：用字符频次作key，避免排序 O(k log k) → O(k)
def groupAnagrams_v2(strs):
    groups = defaultdict(list)
    for s in strs:
        count = [0] * 26
        for c in s:
            count[ord(c) - ord('a')] += 1
        groups[tuple(count)].append(s)  # 频次元组作key
    return list(groups.values())

时间： O(nk)，n为字符串数，k为最长字符串长度；空间： O(nk)

例题2：找到字符串中所有字母异位词（LeetCode 438）— 滑动窗口

题意： 找出s中所有p的字母异位词的起始索引。

from collections import Counter

def findAnagrams(s, p):
    result = []
    if len(s) < len(p): return result

    p_count = Counter(p)             # 模式计数
    window  = Counter(s[:len(p)])    # 初始窗口计数
    if window == p_count:
        result.append(0)

    for i in range(len(p), len(s)):
        # 扩展：加入右侧新字符
        new_char = s[i]
        window[new_char] += 1

        # 收缩：移除左侧旧字符
        old_char = s[i - len(p)]
        window[old_char] -= 1
        if window[old_char] == 0:
            del window[old_char]     # 清理零计数，保证比较正确

        if window == p_count:
            result.append(i - len(p) + 1)
    return result

时间： O(n)；空间： O(Σ)=O(26)

例题3：重复的DNA序列（LeetCode 187）— 滚动哈希

题意： 找出DNA字符串中所有出现超过一次的10个字母的序列。

def findRepeatedDnaSequences(s):
    if len(s) <= 10: return []
    seen = set()
    repeated = set()

    for i in range(len(s) - 9):
        substr = s[i:i+10]
        if substr in seen:
            repeated.add(substr)
        seen.add(substr)
    return list(repeated)

# 滚动哈希版本（O(n)，避免子串切片）
def findRepeatedDnaSequences_hash(s):
    if len(s) <= 10: return []
    BASE = 4
    MAP = {'A': 0, 'C': 1, 'G': 2, 'T': 3}
    MOD = 2**32
    n, L = len(s), 10
    power = BASE**(L-1) % MOD

    h = 0
    for c in s[:L]:
        h = (h * BASE + MAP[c]) % MOD

    seen = {h}
    repeated = set()
    for i in range(1, n - L + 1):
        h = (h - MAP[s[i-1]] * power) % MOD   # 移除旧字符
        h = (h * BASE + MAP[s[i+L-1]]) % MOD  # 加入新字符
        if h in seen:
            repeated.add(s[i:i+L])
        seen.add(h)
    return list(repeated)

时间： O(n)；空间： O(n)

🔴 易错点

① KMP的fail数组回退条件混淆

❌ 错误（回退写成fail[j]）

while j > 0 and p[i] != p[j]:
    j = fail[j]      # 错！应该是fail[j-1]
    # fail[j]是下一个状态，fail[j-1]才是回退目标

✅ 正确

while j > 0 and p[i] != p[j]:
    j = fail[j-1]    # 回退到最长公共前后缀处
if p[i] == p[j]:
    j += 1
fail[i] = j

② 哈希冲突导致误判

❌ 错误（只比较哈希值）

if t_hash == p_hash:
    results.append(i)  # 哈希相等≠字符串相等！
                       # 存在哈希碰撞风险

✅ 正确（哈希+验证）

if t_hash == p_hash and text[i:i+m] == pattern:
    results.append(i)  # 哈希相等 + 逐字符验证

🔷 竞赛拓展

后缀数组（SA）： O(n log n)构建，O(log n)查询任意子串，配合LCP数组解决最长公共子串、字符串排名等高级问题。

AC自动机： 多模式字符串匹配，Trie树 + KMP失配函数，O(n+∑|pattern|+匹配数)时间找出文本中所有模式出现位置。

Manacher算法： O(n)找最长回文子串。通过"镜像"特性跳过已知回文范围内的计算。

def longestPalindrome_manacher(s):
    # 插入分隔符：aba → #a#b#a#
    t = '#' + '#'.join(s) + '#'
    n = len(t)
    p = [0] * n          # p[i] = 以t[i]为中心的最长回文半径
    center = right = 0   # 当前已知最右回文的中心和右边界

    for i in range(n):
        if i < right:
            p[i] = min(right - i, p[2*center - i])  # 镜像初始化
        # 普通扩展
        lo, hi = i - p[i] - 1, i + p[i] + 1
        while lo >= 0 and hi < n and t[lo] == t[hi]:
            p[i] += 1; lo -= 1; hi += 1
        # 更新最右回文
        if i + p[i] > right:
            center, right = i, i + p[i]

    max_len = max(p)
    center_idx = p.index(max_len)
    start = (center_idx - max_len) // 2
    return s[start:start+max_len]
